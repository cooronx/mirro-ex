# 数据库表迁移到本地clickhouse

深圳委托表原表结构

create table L2_order_distributed
(
    channel              Int64,
    message_number       Int64,
    code                 String,
    commission_price     Decimal(20, 4),
    commission_volume    Decimal(20, 4),
    direction            Int8,
    order_type           String,
    commision_time       DateTime64(3),
    extra_message_number Int64,
    EventDate            Date materialized toDate(commision_time)
)
    engine = Distributed('legend_cluster', 'cquote', 'L2_order', intHash64(toUInt64(toDate(commision_time))));

上海委托表原表结构

create table
    L2_shorder_distributed
(
    channel        Int64 comment '频道',
    message_number Int64 comment '序号',
    code           String comment '委托代码',
    time           DateTime64(3) comment '委托时间',
    price          Decimal(20, 4) comment '委托价格，单位: 元',
    volume         Decimal(20, 4) comment '委托量，单位: 股',
    bs_flag        Int8 comment '买卖标记 1: 未知方向, 2: 买, 3: 卖',
    order_type     Int8 comment '委托类型，0: 增加委托单，1: 删除委托单',
    order_number   Int64 comment '委托单号',
    EventDate      Date materialized toDate(time)
)

我们只做名称上面的更改，不做实际的修改

但是有几点需要注意

1. 记得创建表的时候需要创建对应的索引
2. 深圳order删去extra_message_number
3. 上海和深圳的code统一格式，格式为601899.XSHG或者001896.XSHE
4. 字段记得补充注释，注释的内容可以去db文件夹下的common看
